public class SeriesCalculation {
    public static void main(String[] args) {
        int terms = 10000;
        double sum = 0.0;
        
        for (int i = 0; i < terms; i++) {
            double term = (i % 2 == 0 ? 1.0 : -1.0) / (2.0 * i + 1.0);
            sum += term;
        }
        
        double result = sum * 4;
        System.out.printf("The result of the series multiplied by 4 is: %.10f%n", result);
    }
}

Output:
The result of the series multiplied by 4 is: 3.1414926536
